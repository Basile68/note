git branch （查看当前分支名字

1. 查看分支
git branch	                              列出本地所有分支，当前分支带 *	    * main
git branch --show-current	              只显示当前分支名	                main
git status	                              查看当前分支及工作区状态	         On branch main
2. 创建和切换分支
git branch <branch-name>	              创建新分支（不切换）                git branch feature1
git checkout <branch-name>	              切换分支	                          git checkout develop
git switch <branch-name>	              新版本 Git 推荐的切换命令	          git switch main
git checkout -b <branch-name>	         创建并切换分支	                git checkout -b feature/login
git switch -c <branch-name>	              创建并切换分支（新推荐方式）	     git switch -c feature/login
3. 重命名分支
git branch -m <new-name>	              重命名当前分支	                     当前分支改名为 <new-name>
git branch -m <old-name> <new-name>	   重命名指定分支	                     不在当前分支也可重命名
git branch -M <new-name>	              强制重命名当前分支（覆盖同名分支）	比 -m 更强制
git branch -M <old-name> <new-name>	   强制重命名指定分支	                覆盖已有同名分支
4. 删除分支
git branch -d <branch-name>	              删除本地分支（安全删除，必须已合并）	git branch -d feature1
git branch -D <branch-name>	              强制删除本地分支	                 git branch -D feature1
5. 推送分支到远程
git push -u origin <branch-name>	        推送本地分支到远程并设置上游	     第一次推送新分支用
git push origin <branch-name>	        推送本地分支到远程	                已设置上游分支可用
git push origin --delete <branch-name>	   删除远程分支	                      git push origin --delete feature1
